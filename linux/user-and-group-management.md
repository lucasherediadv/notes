```bash
# Add a new user
$ useradd username

# Add user to a supplementary group
$ usermod -aG groupname username

# Delete a user account
$ userdel username

# Set or change password for a user
$ passwd username

# Add a new group
$ groupadd groupname

# Delete a group
$ groupdel groupname

# Display user and gropus IDs
$ id username

# Display your current username
$ whoami

# Switch to another user account
$ su - username

# Run command as superuser (or another user)
$ sudo command
```
